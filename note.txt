# Tentukan folder lokal untuk menyimpan data NLTK
download_dir = 'nltk_data' # root project

# Periksa apakah folder sudah ada
if os.path.exists(download_dir) and os.listdir(download_dir):
    print("Data NLTK sudah ada, tidak perlu mendownload ulang.")

else:
    # Jika folder tidak ada atau kosong, buat folder dan download seluruh data NLTK
    if not os.path.exists(download_dir):
        os.makedirs(download_dir)

    print("Mendownload data NLTK...")
    nltk.download('all', download_dir)
    print("Download selesai.")
# add path for nltk
nltk.data.path.append(download_dir)
